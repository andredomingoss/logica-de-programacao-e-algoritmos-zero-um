Algoritmo "semnome"
// Disciplina   : [Linguagem e Lógica de Programação]
// Autor(a)    : André Domingos
Var
altura, peso, IMC: real


Inicio
escreval("A altura em (m): ")
leia(altura)

     escreval("Parabéns você esta no seu peso ideal! ")
fimse

     escreval("Você não está no seu peso ideal! Você está acima do peso!")
senao
     escreval("Você não está no seu peso ideal! Você está abaixo do peso!")
fimse
